```python
class FourCal:
    def setdata(self, first, second):
        self.first = first
        self.second = second
    
#     def sum(self, a, b):
#         return a+b

    def sum(self):
        result = self.first+self.second
        return result
    
    def mul(self):
        result = self.first * self.second
        return result
    
    def sub(self):
        result = self.first - self.second
        return result

    def div(self):
        result = self.first / self.second
        return result
```
